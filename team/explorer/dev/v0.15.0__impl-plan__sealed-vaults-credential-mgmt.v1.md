## Phase 1: Credential Store

### File: `sg_send_cli/cli/CLI__Credential_Store.py`

```python
import os
from osbot_utils.type_safe.Type_Safe import Type_Safe
from sg_send_cli.secrets.Secrets__Store import Secrets__Store
from sg_send_cli.crypto.Vault__Crypto  import Vault__Crypto

SG_SEND_HOME   = os.path.expanduser('~/.sg-send')
VAULTS_ENC     = os.path.join(SG_SEND_HOME, 'vaults.enc')


class CLI__Credential_Store(Type_Safe):
    secrets : Secrets__Store

    def setup(self):
        """Initialize with default store path."""
        if not self.secrets.store_path:
            self.secrets.store_path = VAULTS_ENC
        return self

    def add_vault(self, passphrase: str, alias: str, vault_key: str):
        self.secrets.store(passphrase, alias, vault_key)

    def get_vault_key(self, passphrase: str, alias: str) -> str:
        return self.secrets.get(passphrase, alias)

    def list_vaults(self, passphrase: str) -> list:
        return self.secrets.list_keys(passphrase)

    def remove_vault(self, passphrase: str, alias: str) -> bool:
        return self.secrets.delete(passphrase, alias)

    def resolve_vault_key(self, alias_or_key: str, passphrase: str = None) -> str:
        """Smart resolution:
        1. If it looks like a vault key (contains ':'), return it directly
        2. Otherwise, treat as alias and look up in credential store
        """
        if ':' in alias_or_key:
            return alias_or_key
        if not passphrase:
            raise RuntimeError(f'Passphrase required to look up vault alias "{alias_or_key}"')
        vault_key = self.get_vault_key(passphrase, alias_or_key)
        if not vault_key:
            raise RuntimeError(f'No vault key found for alias "{alias_or_key}"')
        return vault_key
```

### Passphrase Input

```python
import getpass

def _prompt_passphrase(self, confirm: bool = False) -> str:
    """Prompt for passphrase, optionally with confirmation."""
    passphrase = os.environ.get('SG_SEND_PASSPHRASE')
    if passphrase:
        return passphrase
    passphrase = getpass.getpass('Passphrase: ')
    if confirm:
        confirm_pp = getpass.getpass('Confirm passphrase: ')
        if passphrase != confirm_pp:
            raise RuntimeError('Passphrases do not match')
    return passphrase
```

## Phase 2: Sealed Clone Mode

### File: `sg_send_cli/sync/Vault__Sealed.py`

```python
class Vault__Sealed(Type_Safe):
    crypto : Vault__Crypto

    def is_sealed(self, directory: str) -> bool:
        """Check if .sg_vault/SEALED marker exists."""

    def seal(self, directory: str):
        """Convert regular vault to sealed:
        1. Remove all plaintext files (not in .sg_vault/)
        2. Remove .sg_vault/VAULT-KEY
        3. Create .sg_vault/SEALED marker
        """

    def unseal(self, directory: str, vault_key: str):
        """Convert sealed vault to regular:
        1. Derive read_key from vault_key
        2. Read HEAD commit → tree → decrypt all blobs → write files
        3. Write .sg_vault/VAULT-KEY
        4. Remove .sg_vault/SEALED marker
        """

    def read_file(self, directory: str, vault_key: str, file_path: str) -> bytes:
        """Read a single file from sealed vault:
        1. Derive read_key
        2. Read HEAD → commit → tree
        3. Find blob_id for file_path
        4. Decrypt blob → return plaintext
        All from local object store — no API calls.
        """

    def list_files(self, directory: str, vault_key: str) -> list:
        """List files from sealed vault's tree object."""
```

### Modification to `Vault__Sync.clone()`

Add `sealed` parameter:

```python
def clone(self, vault_key: str, directory: str = None,
          on_progress: callable = None, sealed: bool = False) -> str:
    # ... existing clone logic ...

    # At the end, if sealed:
    if sealed:
        # Don't write plaintext files (skip the file-writing loop's disk writes)
        # Don't write VAULT-KEY
        # Create SEALED marker
        with open(os.path.join(sg_vault_dir, 'SEALED'), 'w') as f:
            f.write('')
    else:
        # Current behavior: write plaintext files + VAULT-KEY
        ...
```

Actually, cleaner approach: **clone always builds the object store**, then:
- Regular mode: also writes plaintext files + VAULT-KEY
- Sealed mode: only writes object store + SEALED marker

## Phase 3: Zip Container

### File: `sg_send_cli/sync/Vault__Zip.py`

```python
import zipfile
import io
from osbot_utils.type_safe.Type_Safe import Type_Safe


class Vault__Zip(Type_Safe):
    crypto : Vault__Crypto

    def pack(self, directory: str, dest_path: str):
        """Pack .sg_vault/ directory into a zip file.
        Entries:
          objects/{prefix}/{object_id}  (encrypted blobs)
          refs/HEAD                     (commit ref)
          tree.json                     (encrypted in sealed mode)
          SEALED                        (marker)
        No VAULT-KEY in the zip. Ever.
        """

    def unpack(self, zip_path: str, directory: str):
        """Unpack zip into .sg_vault/ directory."""

    def read_file(self, zip_path: str, vault_key: str, file_path: str) -> bytes:
        """Read a single file from zip without extracting:
        1. Open zip in memory
        2. Read refs/HEAD → commit object → tree object
        3. Find blob for file_path
        4. Decrypt blob → return plaintext
        """

    def list_files(self, zip_path: str, vault_key: str) -> list:
        """List files from zip's tree object."""

    def from_bytes(self, zip_bytes: bytes, vault_key: str, file_path: str) -> bytes:
        """Read a file from zip bytes in memory.
        For Lambda: load zip from S3 into memory, read a file, done.
        No disk writes at all.
        """
```

### Lambda Deployment Pattern

```python
import boto3
from sg_send_cli.sync.Vault__Zip   import Vault__Zip
from sg_send_cli.crypto.Vault__Crypto import Vault__Crypto

def lambda_handler(event, context):
    # Load encrypted vault zip from S3 (already there as a layer or artifact)
    s3       = boto3.client('s3')
    response = s3.get_object(Bucket='my-configs', Key='deploy.sg-vault.zip')
    zip_data = response['Body'].read()

    # Read a single config file — entirely in memory, no disk writes
    vault_zip  = Vault__Zip(crypto=Vault__Crypto())
    vault_key  = event['vault_key']  # or from Secrets Manager
    config_raw = vault_zip.from_bytes(zip_data, vault_key, 'configs/EC2.json')
    config     = json.loads(config_raw)

    return deploy(config)
```

**This is powerful because:**
- The zip can be stored in S3 alongside Lambda code
- The zip is useless without the vault key
- The vault key can come from AWS Secrets Manager (1 secret vs N config files)
- Zero disk I/O in Lambda (everything in memory)
- Cold start overhead: ~0.1s (unzip in memory + decrypt one file)
