## Phase 1: Credential Store

### File: `sg_send_cli/cli/CLI__Credential_Store.py`

```python
import os
import time
import json
from osbot_utils.type_safe.Type_Safe import Type_Safe
from sg_send_cli.secrets.Secrets__Store import Secrets__Store
from sg_send_cli.crypto.Vault__Crypto  import Vault__Crypto

SG_SEND_HOME   = os.path.expanduser('~/.sg-send')
VAULTS_ENC     = os.path.join(SG_SEND_HOME, 'vaults.enc')
LOCK_STATE     = os.path.join(SG_SEND_HOME, 'lock-state.json')
DEFAULT_LOCK_TIMEOUT = 1800   # 30 minutes of inactivity


class CLI__Credential_Store(Type_Safe):
    secrets      : Secrets__Store
    lock_timeout : int = DEFAULT_LOCK_TIMEOUT   # seconds

    def setup(self):
        if not self.secrets.store_path:
            self.secrets.store_path = VAULTS_ENC
        os.makedirs(SG_SEND_HOME, exist_ok=True)
        return self

    def add_vault(self, passphrase: str, alias: str, vault_key: str):
        self.secrets.store(passphrase, alias, vault_key)
        self._touch_activity()

    def get_vault_key(self, passphrase: str, alias: str) -> str:
        self._check_auto_lock()
        result = self.secrets.get(passphrase, alias)
        self._touch_activity()
        return result

    def list_vaults(self, passphrase: str) -> list:
        self._check_auto_lock()
        result = self.secrets.list_keys(passphrase)
        self._touch_activity()
        return result

    def remove_vault(self, passphrase: str, alias: str) -> bool:
        return self.secrets.delete(passphrase, alias)

    def resolve_vault_key(self, alias_or_key: str, passphrase: str = None) -> str:
        """Smart resolution:
        1. Contains ':' → vault key, return directly
        2. Otherwise → alias, look up in credential store
        """
        if ':' in alias_or_key:
            return alias_or_key
        if not passphrase:
            raise RuntimeError(f'Passphrase required to look up vault alias "{alias_or_key}"')
        vault_key = self.get_vault_key(passphrase, alias_or_key)
        if not vault_key:
            raise RuntimeError(f'No vault key found for alias "{alias_or_key}"')
        return vault_key

    # --- Auto-lock on inactivity ---

    def _touch_activity(self):
        """Record last activity timestamp."""
        state = {'last_activity': time.time()}
        with open(LOCK_STATE, 'w') as f:
            json.dump(state, f)

    def _check_auto_lock(self):
        """Check if credential store should be locked due to inactivity.
        This is checked on every access — passive, no daemon needed."""
        if not os.path.isfile(LOCK_STATE):
            return
        with open(LOCK_STATE, 'r') as f:
            state = json.load(f)
        last = state.get('last_activity', 0)
        if time.time() - last > self.lock_timeout:
            self._lock()

    def _lock(self):
        """Lock the credential store — clear cached passphrase, remove lock state."""
        if os.path.isfile(LOCK_STATE):
            os.remove(LOCK_STATE)
        # Note: since we don't cache passphrases in memory yet (no agent),
        # "locking" just means the next access will require re-entering
        # the passphrase. The auto-lock is future-proofing for when we
        # add passphrase caching (SG_SEND_PASSPHRASE env var or agent).
```

### Auto-Lock Design

The auto-lock is **passive** — no daemon, no background process:
- Every time `sg-send-cli` accesses the credential store, it writes `last_activity` timestamp to `~/.sg-send/lock-state.json`
- Every time `sg-send-cli` accesses the credential store, it first checks if `last_activity` is older than `lock_timeout` (default 30 min)
- If expired: remove lock-state file → next access requires passphrase
- This works because the CLI is the only thing that accesses the store

### Passphrase Input

```python
import getpass

def _prompt_passphrase(self, confirm: bool = False) -> str:
    passphrase = os.environ.get('SG_SEND_PASSPHRASE')
    if passphrase:
        return passphrase
    passphrase = getpass.getpass('Passphrase: ')
    if confirm:
        confirm_pp = getpass.getpass('Confirm passphrase: ')
        if passphrase != confirm_pp:
            raise RuntimeError('Passphrases do not match')
    return passphrase
```

### CLI Commands in `CLI__Vault.py`

```python
def cmd_vault_add(self, args):
    passphrase = self._prompt_passphrase(confirm=True)
    vault_key  = args.vault_key or getpass.getpass('Vault key: ')
    self.credential_store.add_vault(passphrase, args.alias, vault_key)
    print(f'✅ Saved \'{args.alias}\' to ~/.sg-send/vaults.enc')

def cmd_vault_list(self, args):
    passphrase = self._prompt_passphrase()
    aliases    = self.credential_store.list_vaults(passphrase)
    if not aliases:
        print('No stored vaults.')
    else:
        for alias in aliases:
            print(f'  {alias}')

def cmd_vault_remove(self, args):
    passphrase = self._prompt_passphrase()
    removed    = self.credential_store.remove_vault(passphrase, args.alias)
    if removed:
        print(f'Removed \'{args.alias}\'')
    else:
        print(f'No vault found for \'{args.alias}\'')

def cmd_vault_show(self, args):
    passphrase = self._prompt_passphrase()
    vault_key  = self.credential_store.get_vault_key(passphrase, args.alias)
    if vault_key:
        print(vault_key)
    else:
        print(f'No vault found for \'{args.alias}\'')
```

### File: `sg_send_cli/sync/Vault__Bare.py`

```python
class Vault__Bare(Type_Safe):
    crypto : Vault__Crypto

    def is_bare(self, directory: str) -> bool:
        """A vault is bare if .sg_vault/ exists but no VAULT-KEY and no working copy files."""

    def checkout(self, directory: str, vault_key: str):
        """Extract all files from object store to working copy.
        1. Derive read_key from vault_key
        2. Read HEAD → commit → tree from object store
        3. For each entry: load blob → decrypt → write file
        4. Write VAULT-KEY for convenience
        """

    def clean(self, directory: str):
        """Remove working copy files + VAULT-KEY, leaving bare vault.
        1. Read tree to know which files to remove
        2. Remove all listed files
        3. Remove VAULT-KEY, token
        4. Remove empty directories
        """

    def read_file(self, directory: str, vault_key: str, file_path: str) -> bytes:
        """Read one file from bare vault (decrypt from object store)."""

    def list_files(self, directory: str, vault_key: str) -> list:
        """List files from the tree object in the object store."""
```

## Phase 3: Zip Container

### File: `sg_send_cli/sync/Vault__Zip.py`

```python
import zipfile
import io

class Vault__Zip(Type_Safe):
    crypto : Vault__Crypto

    def pack(self, directory: str, dest_path: str):
        """Pack .sg_vault/ into a zip. Never includes VAULT-KEY."""

    def unpack(self, zip_path: str, directory: str):
        """Unpack zip into .sg_vault/ directory."""

    def read_file(self, zip_path: str, vault_key: str, file_path: str) -> bytes:
        """Read a single file from zip without extracting to disk."""

    def list_files(self, zip_path: str, vault_key: str) -> list:
        """List files from zip's tree object."""

    def from_bytes(self, zip_bytes: bytes, vault_key: str, file_path: str) -> bytes:
        """Read a file from zip bytes entirely in memory.
        For Lambda: S3 → bytes → decrypt → config. Zero disk I/O."""

    def from_file(self, zip_path: str, vault_key: str, file_path: str) -> bytes:
        """Read from a zip file on disk (e.g. Lambda layer at /opt/)."""
```
